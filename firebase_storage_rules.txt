rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Existing rules for images and other files
    match /images/{allPaths=**} {
      allow read, write: if request.auth != null;
    }
    
    match /chats/{chatId}/images/{allPaths=**} {
      allow read, write: if request.auth != null;
    }
    
    // New rules for video posts
    match /videos/group_posts/{groupId}/{postId}/{fileName} {
      allow write: if request.auth != null
                   && request.resource.size < 20 * 1024 * 1024
                   && request.resource.contentType.matches('video/.*');
      allow read: if request.auth != null;
    }
    
    match /videos/group_posts/{groupId}/{postId}/thumb.jpg {
      allow write: if request.auth != null
                   && request.resource.size < 2 * 1024 * 1024
                   && request.resource.contentType.matches('image/.*');
      allow read: if request.auth != null;
    }
    
    // Default rule for other files
    match /{allPaths=**} {
      allow read, write: if request.auth != null;
    }
  }
}
